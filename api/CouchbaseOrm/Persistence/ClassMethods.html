def create(attributes = nil, &block)
  if attributes.is_a?(Array)
    attributes.collect { |attr| create(attr, &block) }
  else
    instance = new(attributes, &block)
    instance.save
    instance
  end
end

def create!(attributes = nil, &block)
  if attributes.is_a?(Array)
    attributes.collect { |attr| create!(attr, &block) }
  else
    instance = new(attributes, &block)
    instance.save!
    instance
  end
end

def design_document(name = nil)

  @design_document = name.to_s
end
