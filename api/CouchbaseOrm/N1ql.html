def self.sanitize(value)
  if value.is_a?(String)
    value.gsub("'", "''").gsub('\\'){ '\\\\' }.gsub('"', '\"')
  elsif value.is_a?(Array)
    value.map{ |v| sanitize(v) }
  else
    value
  end
end
